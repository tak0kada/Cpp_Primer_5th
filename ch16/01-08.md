```cpp: compare.hpp
template <typename T>
int compare(const T &lhs, const T &rhs)
{
	if (lhs < rhs)
		return -1;
	else if (lhs > rhs)
		return 1;
	else
		return 0;
}
```

```cpp
/*
namespace std
{
	template < class InputIt, class T >
	InputIt find( InputIt first, InputIt last, const T& value );
}
 */

#include <vector>
#include <string>
#include <list>
#include <iostream>

template <class InputIt, class T>
InputIt find(InputIt first, InputIt last, const T& value)
{
	for (auto pos = first; first != last; ++pos)
	{
		if (*pos == value)
			return pos;
	}
	return last;
}

int main()
{
	using namespace std;

	vector v{1,2,3,4,5};
	list<string> ls{"this", "is", "a", "pen"};

	cout << *find(v.begin(), v.end(), 2) << endl;
	cout << *find(ls.begin(), ls.end(), "pen") << endl;

	return 0;
}
```

# 16.5
```cpp
#include <iostream>
#include <string>


// store information to format matrix output
struct Fmt
{
	std::string end_indent;
};

template <typename T, size_t n>
std::ostream& _print_matrix(std::ostream &os, const T (&arr)[n], const Fmt &fmt)
{
	os << '{';
	for (auto i = 0; i < n - 1; ++i)
	{
		os << arr[i] << ", ";
	}
	os << arr[n - 1] << '}';
	return os;
}

template <typename T, size_t n, size_t m>
std::ostream& _print_matrix(std::ostream &os, const T (&arr)[n][m], const Fmt &fmt)
{
	os << '{';
	for (auto i = 0; i < n - 1; ++i)
	{
		_print_matrix(os, arr[i], Fmt{fmt.end_indent + " "});
		os << ",\n"
		   << fmt.end_indent + " ";
	}
	_print_matrix(os, arr[n - 1], Fmt{fmt.end_indent + " "});
	os << '}';
	return os;
}

template <typename T, size_t n>
void print(const T (&arr)[n])
{
	_print_matrix(std::cout, arr, Fmt{""});
	std::cout << std::endl;
}

template <typename T, size_t n, size_t m>
void print(const T (&arr)[n][m])
{
	_print_matrix(std::cout, arr, Fmt{""});
	std::cout << std::endl;
}


int main()
{
	int arr0[2] = {1,2};
	print(arr0);
	int arr1[2][2] = {{1,2}, {2,3}};
	print(arr1);
	int arr[2][2][2] = {{{-1, 0}, {0, 1}}, {{1, 2}, {2, 3}}};
	print(arr);
}
```
